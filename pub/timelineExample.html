Styled Timeline
                        const timeline = new AdvancedTimeline();

const pointStyle = {
    'background-color': '#0bbb80',
    '--box-shadow-color': '#29d198'
};

const subDivisionStyle = {
    'background-color': '#0bbb80',
    '--box-shadow-color': 'rgb(58 220 165)'
};

const subStyle = {
    'background-color': 'rgb(11, 187, 128)',
    '--box-shadow-color': '#29d198',
    'border-color': '#139113'
};

const [startPoint, endPoint1] = timeline.makeTimeline('#example4', false, 'rgb(64 72 105)',  "Start", "End", 'Starting Point', 'EndPoint', pointStyle, subDivisionStyle, pointStyle);

timeline.makeSubPoint(startPoint, "Point 1 (Click Me)", "Point 1 Information", subStyle);
timeline.makeSubPoint(startPoint, "Point 2", "Point 2 Information", subStyle);

const middlePoint = timeline.makeNewPoint("Middle", 'Middle Point', pointStyle, subDivisionStyle);

timeline.makeSubPoint(middlePoint, "Point 1 (Click Me)", "Point 1 Information", subStyle);
Subpoint Information Card with Text Elements
                        const timeline = new AdvancedTimeline();

const [startPoint, endPoint] = timeline.makeTimeline('#example5', false, 'white',  "Start", "End", 'Starting Point', 'EndPoint');

const subPoint = timeline.makeSubPoint(startPoint, "Point 1 (Click Me)", "Point 1 Information");

//Create a regular text element
timeline.addTextElementToSubPoint(subPoint, 'This is the first subpoint');

//Style a text element
const style = {
    'color': 'green',
    'font-size': '15px'
};

timeline.addTextElementToSubPoint(subPoint, 'This is the first subpoint', '', '', '', '', style);

//Position a text element
timeline.addTextElementToSubPoint(subPoint, 'This is the first subpoint', 50, 20, '', '');
Subpoint Information Card with Image Elements
                        const timeline = new AdvancedTimeline();

const [startPoint, endPoint] = timeline.makeTimeline('#example6', false, 'white',  "Start", "End", 'Starting Point', 'EndPoint');

const subPoint = timeline.makeSubPoint(startPoint, "Point 1 (Click Me)", "Point 1 Information");

//Style an image element
const style = {
    'width': '60px',
    'border': '1px solid orangered',
    "border-radius": '5px'
};

timeline.addImageElementToSubPoint(subPoint, 'timeline.png', '', '', '', '', style);

//Position an image element
timeline.addImageElementToSubPoint(subPoint, 'timeline.png','', 40, '', 0);

Subpoint Information Card with Video Elements
                        const timeline = new AdvancedTimeline();

const [startPoint, endPoint] = timeline.makeTimeline('#example8', false, 'white',  "Start", "End", 'Starting Point', 'EndPoint');

const subPoint = timeline.makeSubPoint(startPoint, "Point 1 (Click Me)", "Point 1 Information");

//Create a regular video element
timeline.addVideoElementToSubPoint(subPoint, 'Timeline.mp4');

//Style a video element
const style = {
    'width': '150px',
    'height': '',
    'border': '1px solid orangered'
};

timeline.addVideoElementToSubPoint(subPoint, 'Timeline.mp4', false, '', '', '', '', style);

//Add video from a different webpage (ex.youtube)
timeline.addVideoElementToSubPoint(subPoint, 'https://www.youtube.com/embed/rTIAcCUaYsw', true, '', '', '', '');

//Position a video element
timeline.addVideoElementToSubPoint(subPoint, 'Timeline.mp4', false, '', 180, '', 30);
